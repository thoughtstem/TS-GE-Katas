11.15 Crafter Kata

read
Make a minecraft game with a basic-skinned hero, carrots, fish, and a crafter on tile 2 (posn 200 200) that makes both carrot-stew and fish-stew.
code
#lang survival-minecraft
 
(define (fish-stew)
   (custom-food
    #:name "Fish Stew"
    #:sprite fish-stew-sprite
    #:respawn? #f
    #:heals-by 50))
 
(define fish-stew-recipe
   (recipe
    #:product (fish-stew)
    #:build-time 40
    #:ingredients (list "Fish")))
 
(define (my-cauldron)
   (custom-crafter
    #:sprite cauldron-sprite
    #:position (posn 200 200)
    #:tile 2
    #:recipe-list (list carrot-stew-recipe fish-stew-recipe)))
 
(minecraft-game
  #:skin (custom-skin)
  #:food-list (list
   (carrot #:amount-in-world 10)
   (fish #:amount-in-world 10))
  #:crafter-list (list (my-cauldron)))

(define (gold-coin)
   (custom-ore

    #:value 40))

    #:sprite toasted-marshmallow-sprite

    #:heals-by 5
    #:amount-in-world 5))
 
(define (cherry)
   (custom-food
    #:sprite cherry-sprite
    #:name "cherry"
    #:heals-by 50
    #:respawn? #f))
 
(define (carrot)
   (custom-food
    #:sprite carrot-sprite
    #:name "carrot"
    #:amount-in-world 5))
 
(define (bowl)
   (custom-food

    #:name "bowl"
    #:heals-by 0
    #:amount-in-world 2))
 
(define (my-enemy-1)
   (custom-mob
    #:ai 'easy
    #:sprite snake-sprite
    #:amount-in-world 5))
 
(define (my-enemy-2)

    #:sprite smores-sprite

    #:name "carrot-stew"
    #:heals-by 60))

    #:product (smores)
    #:build-time 10
    #:ingredients (list "toasted marshmallow")))

    #:product (steak)
    #:build-time 5
    #:ingredients (list "cherry" "toasted marshmallow")))

    #:ingredients (list "carrot" "bowl")))

  #:skin (my-avatar)
  #:ore-list (list (silver-coin) (gold-coin))
